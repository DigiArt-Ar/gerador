FUNÇÃO: GERAR TOKENS COM LÓGICA DE SOMA (MERGE) async function gerarTokens() { const pasta = document.getElementById('pastaNome').value.trim(); const qtd = parseInt(document.getElementById('qtdTokens').value); if (!pasta) { showStatus("Por favor, digite o nome da pasta!", "error"); return; } const dbRef = database.ref('tokens'); // Busca os tokens que já existem para somar const snapshot = await dbRef.once('value'); let tokensAtuais = snapshot.val() || {}; // Gera os novos tokens for (let i = 0; i < qtd; i++) { const novoToken = Math.random().toString(36).substring(2, 8); // Token de 6 caracteres tokensAtuais[novoToken] = { pasta: pasta, status: "disponivel", data_criacao: new Date().toISOString() }; } // Salva o objeto completo (Antigos + Novos) usando .set() após o merge manual // Ou use .update(novosTokens) se preferir a abordagem direta dbRef.set(tokensAtuais) .then(() => { showStatus(`${qtd} tokens gerados para a pasta ${pasta}!`, "success"); }) .catch((error) => showStatus("Erro ao salvar: " + error, "error")); } // 2. FUNÇÃO: RENOVAR TOKEN COM VALIDADE EDITÁVEL function renovarToken() { const token = document.getElementById('tokenRenovar').value.trim(); const dias = parseInt(document.getElementById('validadeDias').value); if (!token) { showStatus("Digite o token para renovar!", "error"); return; } const tokenRef = database.ref(`tokens/${token}`); tokenRef.once('value', (snapshot) => { if (snapshot.exists()) { // Reseta o status e define a nova validade tokenRef.update({ status: "disponivel", // Volta a ficar disponível para o novo ciclo validade_dias: dias, data_renovacao: new Date().toISOString() }).then(() => { showStatus(`Token ${token} renovado por mais ${dias} dias!`, "success"); }); } else { showStatus("Token não encontrado no sistema!", "error"); } }); }
